INSEGUIMENTO TELECAMERA (CORRETTO!) // ========================================== function avviaInseguimento() { function aggiornaPosizioni() { etichetteGenerate.forEach(function(dati) { // La funzione corretta dell'API di Sketchfab: api.getWorldToScreenCoordinates(dati.coord3D, function(risultato) { // Il risultato contiene "canvasCoord" che sono i pixel X e Y esatti if (risultato && risultato.canvasCoord) { var x = risultato.canvasCoord[0]; var y = risultato.canvasCoord[1]; dati.elementoDOM.style.left = x + 'px'; dati.elementoDOM.style.top = y + 'px'; // Nascondi l'etichetta se finisce troppo fuori dallo schermo if (x < -1000 || y < -1000) { dati.elementoDOM.style.opacity = '0'; dati.elementoDOM.style.pointerEvents = 'none'; } else { dati.elementoDOM.style.opacity = '1'; dati.elementoDOM.style.pointerEvents = 'auto'; } } }); }); } // Posiziona i titoli subito aggiornaPosizioni(); // Aggiorna la posizione a ogni movimento api.addEventListener('cameramove', aggiornaPosizioni); }
